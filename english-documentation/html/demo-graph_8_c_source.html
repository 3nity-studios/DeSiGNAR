33 template <class Graph>
34 void write_graph(Graph & g, const char & name)
35 {
36  string connector = "--";
37 
38  if (g.is_digraph())
39  {
40  cout << "di";
41  connector = "->";
42  }
43  cout << "graph " << name << "\n";
44  cout << g.get_num_nodes() << " nodes, " << g.get_num_arcs()
45  << " arcs\n\n";
46 
47  cout << "Node set of " << name << "\n";
48  g.for_each_node([&](const auto & n) { cout << n.get_info() << endl; });
49 
50  cout << "\nArc set of " << name << "\n";
51  g.for_each_arc([&](const auto & a)
52  {
53  const string & s = a.get_src_node().get_info();
54  const string & t = a.get_tgt_node().get_info();
55  cout << s << connector << t << "(" << a.get_info()
56  << ")" << endl;
57  });
58 
59  cout << "\n\n";
60 }

62 template <class Graph>
63 void write_node(Graph & g, typename Graph::Node & p)
64 {
65  string connector = g.is_digraph() ? "->" : "--";
66 
67  g.for_each_adjacent_arc(p, [&](const auto & a)
68  {
69  const string & c =
70  a.get_connected_node(p).get_info();
71  cout << p.get_info() << connector << c
72  << "(" << a.get_info() << ")" << endl;
73  });
74 }
