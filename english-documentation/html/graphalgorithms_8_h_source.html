521  template <class GT>
522  void compute_cut_nodes_rec(GT & g, Node<GT> & p, Arc<GT> & a,
523  SLList<Node<GT> *> & l, lint_t & cdf)
524  {
525  p.visit(GraphTag::DEPTH_FIRST);
526  low<GT>(p) = df<GT>(p) = cdf++;
527 
528  bool is_cut = false;
529 
530  for (AdArcIt<GT> it(g, p); it.has_current(); it.next())
531  {
532  Arc<GT> & arc = it.get_current();
533 
534  if (&arc == &a)
535  continue;
536 
537  Node<GT> & q = it.get_tgt_node();
538 
539  if (q.is_visited(GraphTag::DEPTH_FIRST))
540  {
541  if (not arc.is_visited(GraphTag::DEPTH_FIRST))
542  low<GT>(p) = std::min(low<GT>(p), df<GT>(q));
543 
544  continue;
545  }
546 
547  if (arc.is_visited(GraphTag::DEPTH_FIRST))
548  continue;
549 
550  arc.visit(GraphTag::DEPTH_FIRST);
551 
552  compute_cut_nodes_rec(g, q, arc, l, cdf);
553 
554  low<GT>(p) = std::min(low<GT>(p), low<GT>(q));
555 
556  is_cut = low<GT>(q) >= df<GT>(p) and df<GT>(q) != 0;
557  }
558 
559  if (is_cut)
560  {
561  p.visit(GraphTag::CUT);
562  l.append(&p);
563  }
564 
565  }

568  SLList<Node<GT> *> compute_cut_nodes(GT & g)
569  {

571  {

574  low<GT>(p) = -1;

580

582  start.visit(GraphTag::DEPTH_FIRST);
583  df<GT>(start) = current_df++;
584 
585  SLList<Node<GT> *> list;
586

588  {
589  Node<GT> & t = it.get_tgt_node();

604

606  {
607  start.visit(GraphTag::CUT);
608  list.append(&start);
609  }
610 
611  return list;
