1136  {

1138 
1139  for (auto it = src.nodes_begin(); it != src.nodes_end(); ++it)
1140  {
1141  Node * p = &it.get_current();
1142  Node * q = &tgt.insert_node(p->get_info());
1143  map_nodes[p] = q;
1144  }
1145 
1146  for (auto it = src.arcs_begin(); it != src.arcs_end(); ++it)
1147  {

1149  Node & ssrc = a.get_src_node();
1150  Node & stgt = a.get_tgt_node();
1151  Node & tsrc = *map_nodes[&ssrc];
1152  Node & ttgt = *map_nodes[&stgt];
1153  tgt.insert_arc(tsrc, ttgt, a.get_info());
1154  }
