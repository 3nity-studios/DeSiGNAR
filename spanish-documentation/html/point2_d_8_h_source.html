200  bool is_between(const GenPoint2D & p, const GenPoint2D & q)
201  {
202  if (not is_collinear_with(p, q))
203  return false;
204 
205  if (p.get_x() != q.get_x())
206  return ((p.get_x() <= this->get_x()) and (this->get_x() <= q.get_x()))
207  or ((p.get_x() >= this->get_x()) and (this->get_x() >= q.get_x()));
208  else
209  return ((p.get_y() <= this->get_y()) and (this->get_y() <= q.get_y()))
210  or ((p.get_y() >= this->get_y()) and (this->get_y() >= q.get_y()));
211  }
