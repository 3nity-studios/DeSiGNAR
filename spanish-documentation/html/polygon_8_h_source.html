306  SegmentType get_current()
307  {
308  if (not has_current())
309  throw std::overflow_error("There is not current element");
310 
311  Vertex * next = curr->get_next() == head ?
312  head->get_next() : curr->get_next();
313

315  }

322  Vertex * next = curr->get_next() == head ?
323  head->get_next() : curr->get_next();
324

326  }

328  void next()
329  {
330  if (not has_current())
331  throw std::out_of_range("There is not next element");
332 
333  ++pos;
334  curr = curr->get_next();
335  }

337  void prev()
338  {
339  if (curr == head->get_next())
340  throw std::out_of_range("There is not previous element");
341 
342  --pos;
343  curr = curr->get_prev();
344  }
345 
346  void del()
347  {
348  if (not has_current())
349  throw std::overflow_error("There is not current element");
350 
351  Vertex * c = curr;
352  Vertex * n = nullptr;
353 
354  if (c->get_next() == head)
355  {
356  n = head->get_next();
357  curr = head;
358  }
359  else
360  {
361  next();
362  n = curr;
363  next();
364  }
365 
366  p_ptr->num_vertices -= 2;
367 
368  c->del();
369  n->del();
370 
371  delete c;
372  delete n;
373  }

429  void GenPolygon<PointT>::copy(const GenPolygon & p)
430  {
431  Vertex * v = dl_to_vertex(p.vertex_list.get_next());
432 
433  while (v != &p.vertex_list)
434  {
435  vertex_list.insert_prev(new Vertex(v->get_item()));
