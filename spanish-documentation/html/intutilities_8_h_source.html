47  template <typename T>
48  T forward_prod(T a, T b)
49  {
50  static_assert(std::is_integral<T>::value,
51  "Argument must be an integral type");
52 
53  T ret_val = 1;
54 
55  while (a <= b)
56  ret_val *= a++;
57 
58  return ret_val;
59  }
