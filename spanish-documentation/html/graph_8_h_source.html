732  template <typename GT, class Node, class Arc>
733  void BaseGraph<GT, Node, Arc>::copy_graph(const GT & src, GT & tgt)
734  {
735  HashMap<Node *, Node *> map_nodes;
736 
737  for (auto it = src.nodes_begin(); it != src.nodes_end(); ++it)
738  {
739  Node * p = &it.get_current();
740  Node * q = &tgt.insert_node(p->get_info());
741  map_nodes[p] = q;
742  }
743 
744  for (auto it = src.arcs_begin(); it != src.arcs_end(); ++it)
745  {
746  Arc & a = it.get_current();
747  Node & ssrc = a.get_src_node();
748  Node & stgt = a.get_tgt_node();
749  Node & tsrc = *map_nodes[&ssrc];
750  Node & ttgt = *map_nodes[&stgt];
751  tgt.insert_arc(tsrc, ttgt, a.get_info());
752  }
