194  real_t slope() const
195  {
196  if (tgt_point.get_x() == src_point.get_x())
197  {
198  if (src_point.get_y() < tgt_point.get_y())

218  PointT intersection_with(const GenSegment & s) const
219  {
220  if (is_parallel_with(s))
221  throw std::domain_error("Segments are parallels");
222 
223  const typename PointT::NumberType & x1 = src_point.get_x();
224  const typename PointT::NumberType & y1 = src_point.get_y();
225 
226  const typename PointT::NumberType & x2 = s.src_point.get_x();
227  const typename PointT::NumberType & y2 = s.src_point.get_y();
228 
229  const real_t & m1 = slope();
230  const real_t & m2 = s.slope();
231 
232  const typename PointT::NumberType x =
233  real_t(y2 - y1 + m1 * x1 - m2 * x2) / real_t(m1 - m2);
234 
235  const typename PointT::Number_Type y = m1 * (x - x1) + y1;
236 
237  return PointT(x, y);
238  }
