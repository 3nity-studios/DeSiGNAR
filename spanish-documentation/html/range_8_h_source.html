34  template <typename T>
35  class Range
36  {
37  static_assert(std::is_arithmetic<T>::value,

44  public:
45  Range(T _first, T _last, T _step = T(1))
46  : __first(_first), __last(_last), __step(_step)
47  {
48  if (__first > __last)
49  throw std::range_error("First value cannot be greater than last value");
50
