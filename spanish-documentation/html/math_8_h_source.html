137  template <typename BT, typename ET>
138  BT fast_integral_pow(BT base, ET exp)
139  {
140  static_assert(std::is_integral<BT>::value and std::is_integral<ET>::value,
141  "Arguments must be integral types");
142  static_assert(std::is_unsigned<ET>::value,
143  "Exponent must be an unsigned type");
144 
145  BT ret_val = BT(1);
146 
147  while (exp > ET(0))
148  {
149  if (exp & ET(1))
150  ret_val *= base;
151 
152  exp >>= 1;
153  base *= base;
154  }
155 
156  return ret_val;
157  }
